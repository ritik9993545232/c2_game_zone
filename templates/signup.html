// In your signup form JavaScript fetch('/signup', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ username: username, email: email, password: password }) }) .then(response => response.json()) .then(data => { if (data.success) { // Show success message alert(data.message); // Redirect to login page if (data.redirect) { window.location.href = data.redirect; } else { window.location.href = '/login'; } } else { alert(data.error); } }) .catch(error => { console.error('Error:', error); alert('An error occurred during signup'); });
